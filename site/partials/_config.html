Configuration

Drop this MCP config into your OpenCode settings.

{
  "mcp": {
    "nano-brain": {
      "type": "local",
      "command": ["node", "path/to/nano-brain/bin/cli.js", "mcp"],
      "enabled": true
    }
  }
}
# ~/.config/nano-brain/config.yml
collections:
  memory:
    path: ~/.nano-brain
    pattern: "**/*.md"
    update: auto
  sessions:
    path: ~/.local/share/opencode/sessions
    pattern: "**/*.json"
    update: auto
